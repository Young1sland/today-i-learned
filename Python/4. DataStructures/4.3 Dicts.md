```py
# dict는 수정 가능함.
player = {
  'name': 'nico',
  'age': 12,
  'alive': True,
  'fav_food': ["🍕", "🍔"]
}
# key로 조회
print(player.get('age')) #12
print(player.get('fav_food')) # ['🍕', '🍔']

# 삭제
player.pop('age') #age key,value 삭제
print(player) # {'name': 'nico', 'alive': True, 'fav_food': ['🍕', '🍔']}

# 추가
player['xp'] =1500
print(player) # {'name': 'nico', 'alive': True, 'fav_food': ['🍕', '🍔'], 'xp': 1500}

player['fav_food'].append("🍜") 
print(player.get('fav_food')) # ['🍕', '🍔', '🍜']
print(player['fav_food']) # ['🍕', '🍔', '🍜']

```
