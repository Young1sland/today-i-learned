함수가 데이터와 결합되어 있다면 메소드라 부르고 그렇지 않다면 함수라 부름.
```py
# "nico"데이터와 결합되어 있으므로 메소드라 부름
print("nico".endswith("o")) #True
```

```py
# Lists
days_of_week = ["Mon", "Tue", "Wed", "Thur", "Fri"]

# Count
print(days_of_week.count("Wed"))  # 1

# Reverse
days_of_week.reverse()
print(days_of_week)  # ['Fri', 'Thur', 'Wed', 'Tue', 'Mon']

# Append, Remove
days_of_week.append("Sat")
days_of_week.append("Sun")
days_of_week.remove("Sun")
print(days_of_week)  # ['Fri', 'Thur', 'Wed', 'Tue', 'Mon', 'Sat']

# index 접근
print(days_of_week[2])  # Wed

# Clear
days_of_week.clear()
print(days_of_week)  # []
```
