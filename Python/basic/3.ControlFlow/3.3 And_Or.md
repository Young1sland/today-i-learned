```py
#입력받기
age = input("How old are you?")
print("user answer", type(age))
#> <class 'str'>

#값을 integer로 변경
age = int(input("How old are you?"))
print(type(age))
#> <class 'int'>

#and,or
if age < 18:
    print("You can't drink.")
elif age > 18 and age <= 35:
    print("You drink beer")
elif age == 60 or age == 70:
    print("Birthday party!")
else:
    print("Go ahead!")
```
