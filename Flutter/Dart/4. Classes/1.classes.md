class Player {
  late final String name;
  late final int xp;

  /* Player(String name, int xp) {
    this.name = name;
    this.xp = xp;
  }
 */
  Player(this.name, this.xp); //위의 생성자를 이렇게 한줄로 표현 가능

  void sayHello() {
    //this.name 사용 가능하나 이름이 겹치는 경우가 아니면 class안에서는 쓰지 말자
    print("Hi my name is $name");
  }
}

void main() {
  var player = Player("wisdom", 1500);
  player.sayHello();
  var player2 = Player("Kim", 2500); //positional syntax, default: required
  player2.sayHello();
}
