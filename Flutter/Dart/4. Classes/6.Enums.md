enum XPLevel {beginner, medium}

class Player {
  String name;
  XPLevel xp;
  Team team;
  Player({
    required this.name,
    required this.xp,
    required this.team,
  });

  void sayHello() {
    print("Hi my name is $name");
  }
}

void main() {
  var wisdom = Player(name: 'wisdom', xp: 12, team: 'red')
